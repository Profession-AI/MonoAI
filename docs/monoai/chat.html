175    def _initialize_or_create_chat(self, chat_id: str, system_prompt: str) -> str:
176        """Initialize existing chat or create new one with specified ID.
177        
178        Parameters
179        ----------
180        chat_id : str
181            The chat ID to check/create
182        system_prompt : str
183            System prompt to use if creating new chat
184            
185        Returns
186        -------
187        str
188            The chat ID (either existing or newly created)
189        """
190        try:
191            # Try to load existing chat
192            existing_messages = self._history.load(chat_id)
193            
194            # Check if chat exists and has messages
195            if existing_messages and len(existing_messages) > 0:
196                logger.info(f"Loaded existing chat: {chat_id}")
197                return chat_id
198            else:
199                # Chat doesn't exist, create it with the specified ID
200                logger.info(f"Creating new chat with ID: {chat_id}")
201                return self._create_chat_with_id(chat_id, system_prompt)
202                
203        except Exception as e:
204            # If loading fails, assume chat doesn't exist and create it
205            logger.info(f"Failed to load chat {chat_id}, creating new one: {e}")
206            return self._create_chat_with_id(chat_id, system_prompt)

261    def _process_system_prompt(self, system_prompt: Optional[Union[Prompt, str]]) -> str:
262        """Process and load the system prompt."""
263        try:
264            if system_prompt is None:
265                return self._load_default_system_prompt()
266            elif isinstance(system_prompt, str) and system_prompt.endswith(".prompt"):
267                return self._load_prompt_file(system_prompt)
268            elif isinstance(system_prompt, Prompt):
269                return str(system_prompt)

306    def _process_file_attachment(self, prompt: str, file: Optional[Union[str, bytes]], file_type: Optional[str]) -> Union[str, List[Dict[str, Any]]]:
307        """Process file attachment and return modified prompt."""
308        if file is None:
309            return prompt
310            
311        try:
312            # Determine file type
313            if file_type is None and isinstance(file, str):
314                file_type = file.split(".")[-1].lower()
315                
316            if not file_type:
317                raise ChatError("File type could not be determined")
318                
319            conf = Conf()
320            
321            # Handle text files
322            if file_type in conf["supported_files"]["text"]:
323                return self._process_text_file(prompt, file)
324            # Handle image files
325            elif file_type in conf["supported_files"]["image"]:
326                return self._process_image_file(prompt, file)
327            else:
328                raise ChatError(f"Unsupported file type: {file_type}")
329                
330        except Exception as e:
331            raise ChatError(f"Failed to process file attachment: {e}")

333    def _process_text_file(self, prompt: str, file: Union[str, bytes]) -> str:
334        """Process text file attachment."""
335        try:
336            if isinstance(file, str):
337                # Handle as file path or URL
338                if os.path.exists(file):
339                    with open(file, "r", encoding="utf-8") as f:
340                        file_content = f.read()
341                else:
342                    # Assume it's a URL or remote file
343                    file_content = file
344            else:
345                # Handle as bytes
346                file_content = file.decode("utf-8")
347                
348            return prompt + Conf()["default_prompt"]["file"] + file_content
349            
350        except Exception as e:
351            raise ChatError(f"Failed to process text file: {e}")

353    def _process_image_file(self, prompt: str, file: Union[str, bytes]) -> List[Dict[str, Any]]:
354        """Process image file attachment."""
355        try:
356            if isinstance(file, str):
357                # Handle as file path or URL
358                if os.path.exists(file):
359                    # Convert local file to base64
360                    with open(file, "rb") as f:
361                        file_bytes = f.read()
362                    image_url = f"data:image/jpeg;base64,{base64.b64encode(file_bytes).decode('utf-8')}"
363                else:
364                    # Assume it's already a URL
365                    image_url = file
366            else:
367                # Handle as bytes
368                image_url = f"data:image/jpeg;base64,{base64.b64encode(file).decode('utf-8')}"
369                
370            return [
371                {"type": "text", "text": prompt},
372                {"type": "image_url", "image_url": image_url}
373            ]
374            
375        except Exception as e:
376            raise ChatError(f"Failed to process image file: {e}")

378    def _prepare_messages(self, prompt: Union[str, List[Dict[str, Any]]]) -> List[Dict[str, Any]]:
379        """Prepare messages for the API call."""
380        try:
381            messages = self._history.load(self.chat_id)
382            messages = [{'role': d.get('role'), 'content': d.get('content')} for d in messages]
383
384            messages.append({"role": "user", "content": prompt})
385
386            # Apply history summarization if enabled
387            if self._history_summarizer is not None:
388                return self._apply_history_summarization(messages)
389            else:
390                return messages
391                
392        except Exception as e:
393            raise ChatError(f"Failed to prepare messages: {e}")
394
395    def _apply_history_summarization(self, messages: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
396        """Apply history summarization to messages."""
397        try:
398            summarized = self._history_summarizer.summarize(messages)
399            ask_messages = [messages[0], messages[-1]]
400            ask_messages[0]["content"] += Conf()["default_prompt"]["summary"] + summarized
401            return ask_messages
402        except Exception as e:
403            logger.warning(f"History summarization failed, using full history: {e}")
404            return messages

406    def _store_messages(self, messages: List[Dict[str, Any]], response_content: str) -> None:
407        """Store messages in history."""
408        try:
409            messages.append({"role": "assistant", "content": response_content})
410            self._history.store(self.chat_id, messages[-2:])
411        except Exception as e:
412            logger.error(f"Failed to store messages: {e}")

415    def ask(self, prompt: str, file: Optional[Union[str, bytes]] = None, file_type: Optional[str] = None, return_history: bool = False, metadata: Optional[Dict[str, Any]] = {}) -> Union[str, List[Dict[str, Any]]]:
416        """
417        Ask the model a question.
418
419        Parameters
420        ----------
421        prompt : str
422            The question to ask the model
423        file : str | bytes, optional
424            The file to attach to the message, if it's a string it will be treated as a local path or remote url to a file
425        file_type : str, optional
426            The type of the file
427        return_history : bool, optional
428            Whether to return the full history of the chat or only the response
429        metadata : Dict[str, Any], optional
430            Metadata to pass to the model for observability and tracking
431
432        Returns
433        -------
434        str | List[Dict[str, Any]]
435            The response from the model or the full history if return_history is True
436
437        Raises
438        ------
439        ChatError
440            If the request fails or parameters are invalid
441        """
442
443        metadata = self._metadata | metadata
444
445        try:
446            # Process file attachment
447            processed_prompt = self._process_file_attachment(prompt, file, file_type)
448            
449            # Prepare messages
450            messages = self._prepare_messages(processed_prompt)
451            
452            # Make API call
453            response = self._model.ask(messages, metadata=metadata)
454            response_content = response["response"]
455            
456            # Store messages
457            self._store_messages(messages, response_content)
458            
459            response = {
460                "chat_id": self.chat_id,
461                "response": response_content,
462                "model": {
463                    "provider": self._model.provider,
464                    "name": self._model.model
465                }
466            }
467
468            if return_history:
469                response["history"] = messages
470            
471            return response
472                
473        except Exception as e:
474            logger.error(f"Ask request failed: {e}")
475            raise ChatError(f"Ask request failed: {e}")

478        """
479        Ask the model a question and stream the response.
480
481        Parameters
482        ----------
483        prompt : str
484            The question to ask the model
485        file : str | bytes, optional
486            The file to attach to the message
487        file_type : str, optional
488            The type of the file
489        metadata : Dict[str, Any], optional
490            Metadata to pass to the model for observability and tracking
491
492        Yields
493        ------
494        str
495            JSON string containing streaming response chunks
496
497        Raises
498        ------
499        ChatError
500            If the request fails or parameters are invalid
501        """
502
503        metadata = self._metadata | metadata
504
505        try:
506            # Process file attachment
507            processed_prompt = self._process_file_attachment(prompt, file, file_type)

541            Metadata to pass to the model for observability and tracking

548        Raises

575                response_text += content
576                yield content
577

579            self._store_messages(messages, response_text)
